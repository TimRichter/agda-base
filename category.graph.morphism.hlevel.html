morphism-structure-iso : ∀ {i₁ j₁ i₂ j₂}
                         {W : Graph i₁ j₁}
                         {U : Graph i₂ j₂}
                       → ( Σ (∣ W ∣ → ∣ U ∣) λ f
                         → IsMorphism {W = W}{U = U} f )
                       ≅ Morphism W U
morphism-structure-iso = bundle-structure-iso IsMorphism
