unit-groupoid : Groupoid _ _
unit-groupoid = discrete (⊤ , h! ⊤-contr)

unit : Category _ _
unit = Groupoid.cat unit-groupoid

unit-func : ∀ {i j}(C : Category i j) → Functor C unit
unit-func C = Const C tt
